#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;

    while (t--) {
        int n, k;
        cin >> n >> k;

        vector<int> a(n), b;
        for (int i = 0; i < n; i++) {
            cin >> a[i];
        }

        if (k > 1) {
            cout << "YES\n";
            continue;
        }

        b = a;
        sort(b.begin(), b.end());

        if (a == b)
            cout << "YES\n";
        else
            cout << "NO\n";
    }

    return 0;
}
